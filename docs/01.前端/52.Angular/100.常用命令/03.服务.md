## 创建服务

```
ng g service services/storage     //创建到指定目录下
```



## app.module.ts引入服务

```typescript
import {StorageService} from './services/storage.service';//引入

@NgModule({
  declarations: [
    AppComponent,
    ConditionComponent
  ],
  imports: [
    BrowserModule,
    AppRoutingModule,
    AgGridModule.withComponents([]),
    FormsModule

  ],
  providers: [StorageService],//声明
  bootstrap: [AppComponent]
```
